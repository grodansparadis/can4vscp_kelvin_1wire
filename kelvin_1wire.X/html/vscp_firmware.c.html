115   //  vscp_init
116   //
117  
118   void  vscp_init(void)
119   {
120        vscp_initledfunc  =  VSCP_LED_ON;
121  
122        //  read  the  nickname  id
123        vscp_nickname  =  vscp_readNicknamePermanent();
124  
125        //  if  zero  set  to  uninitialised
126        if  (  0  ==  vscp_nickname)  vscp_nickname  =  VSCP_ADDRESS_FREE;
127  
128        //  Start  init.  blink  if  no  nodeid
129        if  (  VSCP_ADDRESS_FREE  ==  vscp_nickname  )  vscp_initledfunc  =  VSCP_LED_BLINK1;
130  
131        //  Init.  incoming  event
132        vscp_imsg.flags  =  0;
133        vscp_imsg.priority  =  0;
134        vscp_imsg.vscp_class  =  0;
135        vscp_imsg.vscp_type  =  0;
136  
137        //  Init.  outgoing  event
138        vscp_omsg.flags  =  0;
139        vscp_omsg.priority  =  0;
140        vscp_omsg.vscp_class  =  0;
141        vscp_omsg.vscp_type  =  0;
142  
143        vscp_errorcnt  =  0;       //  No  errors  yet
144        vscp_alarmstatus  =  0;    //  No  alarm  status
145  
146        vscp_probe_address  =  0;
147  
148        //  Initial  state
149        vscp_node_state  =  VSCP_STATE_STARTUP;
150        vscp_node_substate  =  VSCP_SUBSTATE_NONE;
151  
152        vscp_probe_cnt  =  0;
153        vscp_page_select  =  0;
154  
155        //  Initialise  time  keeping
156        vscp_timer  =  0;
157        vscp_configtimer  =  0;
158        vscp_second  =  0;
159        vscp_minute  =  0;
160        vscp_hour  =  0;
161   }

163   ///////////////////////////////////////////////////////////////////////////////
164   //  vscp_check_pstorage
165   //
166   //  Check  control  position  integrity  and  restore  EEPROM
167   //  if  invalid.
168   //
169  
170   int8_t  vscp_check_pstorage(void)
171   {
172        //  Check  if  persistent  storage  is  initialised.
173        if  (  (  VSCP_INITIALIZED_BYTE0_VALUE  ==
174                        vscp_getControlByte(  VSCP_INITIALIZED_BYTE0_INDEX  )  )  &&
175               (  VSCP_INITIALIZED_BYTE1_VALUE  ==
176                        vscp_getControlByte(  VSCP_INITIALIZED_BYTE1_INDEX  )  )  )  {
177            return  TRUE;
178        }
179  
180        vscp_initledfunc  =  VSCP_LED_BLINK1;
181  
182        //  Initialise  persistent  storage
183        vscp_init_pstorage();
184  
185        //  No  nickname  yet.
186        vscp_writeNicknamePermanent(  VSCP_ADDRESS_FREE  );
187  
188        //  Mark  persistent  storage  as  initialised
189        vscp_setControlByte(  VSCP_INITIALIZED_BYTE0_INDEX,
190                                VSCP_INITIALIZED_BYTE0_VALUE  );
191        vscp_setControlByte(  VSCP_INITIALIZED_BYTE1_INDEX,
192                                VSCP_INITIALIZED_BYTE1_VALUE  );
193  
194        return  FALSE;
195   }

206   ///////////////////////////////////////////////////////////////////////////////
207   //  vscp_handleProbe
208   //
209  
210   void  vscp_handleProbeState(void)
211   {
212        switch  (vscp_node_substate)  {
213  
214            case  VSCP_SUBSTATE_NONE:
215  
216                if  (VSCP_ADDRESS_FREE  !=  vscp_probe_address)  {
217  
218                    vscp_omsg.flags  =  VSCP_VALID_MSG  +  1;  //  one  data  byte
219                    vscp_omsg.priority  =  VSCP_PRIORITY_HIGH;
220                    vscp_omsg.vscp_class  =  VSCP_CLASS1_PROTOCOL;
221                    vscp_omsg.vscp_type  =  VSCP_TYPE_PROTOCOL_NEW_NODE_ONLINE;
222                    vscp_omsg.data[  0  ]  =  vscp_probe_address;
223  
224                    //  send  the  probe
225                    vscp_sendEvent();
226  
227                    vscp_node_substate  =  VSCP_SUBSTATE_INIT_PROBE_SENT;
228                    vscp_timer  =  0;
229  
230                }
231                else  {
232  
233                    //  No  free  address  -&rsaquo   error
234                    vscp_node_state  =  VSCP_STATE_ERROR;
235  
236                    //  Tell  system  we  are  giving  up
237                    vscp_omsg.flags  =  VSCP_VALID_MSG  +  1;    //  one  data  byte
238                    vscp_omsg.data[  0  ]  =  0xff;              //  we  are  unassigned
239                    vscp_omsg.priority  =  VSCP_PRIORITY_LOW;
240                    vscp_omsg.vscp_class  =  VSCP_CLASS1_PROTOCOL;
241                    vscp_omsg.vscp_type  =  VSCP_TYPE_PROTOCOL_PROBE_ACK;
242  
243                    //  send  the  error  event
244                    vscp_sendEvent();
245  
246                }
247                break;
248  
249            case  VSCP_SUBSTATE_INIT_PROBE_SENT:
250  
251                if  (vscp_imsg.flags  &  VSCP_VALID_MSG)  {  //  incoming  event?
252  
253                    //  Yes,  incoming  event
254                    if  ((VSCP_CLASS1_PROTOCOL  ==  vscp_imsg.vscp_class)  &&
255                            (VSCP_TYPE_PROTOCOL_PROBE_ACK  ==  vscp_imsg.vscp_type))  {
256  
257                        //  Yes  it  was  an  ack  from  the  segment  master  or  a  node
258                        if  (0  ==  vscp_probe_address)  {
259  
260                            //  Master  controller  answered
261                            //  wait  for  address
262                            vscp_node_state  =  VSCP_STATE_PREACTIVE;
263                            vscp_timer  =  0;  //  reset  timer
264  
265                        }
266                        else  {
267  
268                            //  node  answered,  try  next  address
269                            vscp_probe_address++;
270                            vscp_node_substate  =  VSCP_SUBSTATE_NONE;
271                            vscp_probe_cnt  =  0;
272  
273                        }
274                    }
275                }
276                else  {
277  
278                    if  (vscp_timer &rsaquo   VSCP_PROBE_TIMEOUT)  {  //  Check  for  time-out
279  
280                        vscp_probe_cnt++;  //  Another  time-out
281  
282                        if  (vscp_probe_cnt &rsaquo =  VSCP_PROBE_TIMEOUT_COUNT)  {
283  
284                            //  Yes  we  have  a  time-out
285                            if  (0  ==  vscp_probe_address)  {  //  master  controller  probe?
286  
287                                //  No  master  controller  on  segment,  try  next  node
288                                vscp_probe_address++;
289                                vscp_node_substate  =  VSCP_SUBSTATE_NONE;
290                                vscp_timer  =  0;
291                                vscp_probe_cnt  =  0;
292  
293                            }
294                            else  {
295  
296                                //  We  have  found  a  free  address  -  use  it
297                                vscp_nickname  =  vscp_probe_address;
298                                vscp_node_state  =  VSCP_STATE_ACTIVE;
299                                vscp_node_substate  =  VSCP_SUBSTATE_NONE;
300                                vscp_writeNicknamePermanent(vscp_nickname);
301  
302                                //  Report  success
303                                vscp_probe_cnt  =  0;
304                                vscp_goActiveState();
305  
306                            }
307                        }
308                        else  {
309                            vscp_node_substate  =  VSCP_SUBSTATE_NONE;
310                        }
311                    }  //  Timeout
312  
313                }
314                break;
315  
316            case  VSCP_SUBSTATE_INIT_PROBE_ACK:
317                break;
318  
319            default:
320                vscp_node_substate  =  VSCP_SUBSTATE_NONE;
321                break;
322        }
323  
324        vscp_imsg.flags  =  0;
325  
326   }

329   //  vscp_handlePreActiveState
330   //
331  
332   void  vscp_handlePreActiveState(void)
333   {
334  
335        if  (  vscp_imsg.flags  &  VSCP_VALID_MSG  )  {  //  incoming  event?
336  
337            if  ((VSCP_CLASS1_PROTOCOL  ==  vscp_imsg.vscp_class)  &&
338                    (VSCP_TYPE_PROTOCOL_SET_NICKNAME  ==  vscp_imsg.vscp_type)  &&
339                    (VSCP_ADDRESS_FREE  ==  vscp_imsg.data[  0  ]))  {
340  
341                //  Assign  nickname
342                vscp_nickname  =  vscp_imsg.data[  1  ];
343                vscp_writeNicknamePermanent(vscp_nickname);
344  
345                //  Go  active  state
346                vscp_node_state  =  VSCP_STATE_ACTIVE;
347            }
348        }
349        else  {
350            //  Check  for  time  out
351            if  (vscp_timer &rsaquo   VSCP_PROBE_TIMEOUT)  {
352                //  Yes,  we  have  a  timeout
353                vscp_nickname  =  VSCP_ADDRESS_FREE;
354                vscp_writeNicknamePermanent(VSCP_ADDRESS_FREE);
355                vscp_init();
356            }
357        }
358   }

410   //  vscp_handleSetNickname
411   //
412  
413   void  vscp_handleSetNickname(void)
414   {
415        if  (  (  2  ==  (vscp_imsg.flags  &  0x0f  )  )  &&
416                (vscp_nickname  ==  vscp_imsg.data[  0  ]))  {
417  
418            //  Yes,  we  are  addressed
419            vscp_nickname  =  vscp_imsg.data[  1  ];
420            vscp_writeNicknamePermanent(vscp_nickname);
421  
422                   //return  nickname  accepted
423                   vscp_omsg.flags  =  VSCP_VALID_MSG  +  1;  //  one  data  byte
424                   vscp_omsg.priority  =  VSCP_PRIORITY_HIGH;
425                   vscp_omsg.vscp_class  =  VSCP_CLASS1_PROTOCOL;
426                   vscp_omsg.vscp_type  =  VSCP_TYPE_PROTOCOL_NICKNAME_ACCEPTED;
427                   vscp_omsg.data[  0  ]  =  vscp_nickname;
428  
429                   //  send  the  event
430                   vscp_sendEvent();
431  
432        }
433   }

435   ///////////////////////////////////////////////////////////////////////////////
436   //  vscp_handleDropNickname
437   //
438  
439   void  vscp_handleDropNickname(void)
440   {
441        uint8_t  bytes  =  vscp_imsg.flags  &  0x0f;
442  
443        #ifdef  DROP_NICKNAME_EXTENDED_FEATURES
444        uint8_t  brake  =  0;
445        #endif
446  
447        if  ((bytes &rsaquo =  1)  &&  (vscp_nickname  ==  vscp_imsg.data[  0  ]))  {
448            //  Yes,  we  are  addressed
449  
450   #ifdef  DROP_NICKNAME_EXTENDED_FEATURES
451            //  Optional  Byte  1:
452            //  bit7  -  go  idle  do  not  start,  bit6  -  reset  persistent  storage
453            //  bit5  -  reset  device  but  keep  nickname
454            //  bit5  and  bit  7  are  concurrent,  here  I  give  bit  5  higher  priority
455            //  Optional  byte  2:  time  in  seconds  before  restarting  (makes  only  sense
456            //  with  either  none  of  the  bits  or  only  bit  5  of  byte1  set.
457            if  (bytes &rsaquo =  2)  {   //  byte  1  does  exist
458                //  bit  6  set:  reset  persistent  storage,  continue  to  check  other
459                //  options  in  byte  1
460                if  (vscp_imsg.data[1]  &  (1&lsaquo &lsaquo 6))  {
461                    //  reset  persistent  storage  here
462                }
463                //  bit  5  set:  reset  device,  keep  nickname,  disregard  other  option
464                //  below  this  by  using  'brake'
465                if  ((vscp_imsg.data[1]  &  (1&lsaquo &lsaquo 5))  &&  (brake  ==  0))  {
466                    vscp_hardreset();
467                    brake  =  1;
468                }
469  
470                //  bit  7  set:  go  idle,  e.g.  stay  in  an  endless  loop  until  re-power
471                if  ((vscp_imsg.data[1]  &  (1&lsaquo &lsaquo 7))  &&  (brake  ==  0))  {
472                    vscp_nickname  =  VSCP_ADDRESS_FREE;
473                    vscp_writeNicknamePermanent(VSCP_ADDRESS_FREE);
474                    for  (;;)  {};  //  wait  forever
475                }
476            }
477   #endif
478            //  none  of  the  options  from  byte  1  have  been  used  or  byte  1  itself
479            //  has  not  been  transmitted  at  all
480            if  ((bytes  ==  1)  ||  ((bytes &rsaquo   1)  &&  (vscp_imsg.data[1]  ==  0)))  {
481                //  this  is  the  regular  behaviour  without  using  byte  1  options
482                vscp_nickname  =  VSCP_ADDRESS_FREE;
483                vscp_writeNicknamePermanent(VSCP_ADDRESS_FREE);
484                vscp_init();
485            }
486   #ifdef  DROP_NICKNAME_EXTENDED_FEATURES
487            //  now  check  if  timing  was  passed  in  byte  2
488            if  (bytes &rsaquo   2)  {
489                //  and  waiting  for  options  that  made  sense
490                if  (  (vscp_imsg.data[1]  ==  0)  ||
491                        (  vscp_imsg.data[1]  &  (1&lsaquo &lsaquo 6))  ||
492                        (  vscp_imsg.data[1]  &  (1&lsaquo &lsaquo 5))  )  {
493                    //  wait  platform  independently
494                    vscp_wait_s(vscp_imsg.data[2]);
495                }
496            }
497   #endif
498        }
499   }

501   ///////////////////////////////////////////////////////////////////////////////
502   //  vscp_newNodeOnline
503   //
504  
505   void  vscp_newNodeOnline(void)
506   {
507        if  (  (  1  ==  (  vscp_imsg.flags  &  0x0f  )  )  &&
508                (vscp_nickname  ==  vscp_imsg.data[  0  ]))  {
509  
510            //  This  is  a  probe  which  use  our  nickname
511            //  we  have  to  respond  to  tell  the  new  node  that
512            //  this  nickname  is  in  use
