152   void  interrupt  low_priority  interrupt_at_low_vector(  void  )
153   {
154        //  Check  timer
155        if  (INTCONbits.TMR0IF)  {  //  If  A  Timer0  Interrupt,  Then
156  
157            //  Reload  value  for  1  ms  resolution
158            WriteTimer0(TIMER0_RELOAD_VALUE);
159  
160            vscp_timer++;
161            vscp_configtimer++;
162            measurement_clock++;
163            timeout_clock++;
164            sendTimer++;
165  
166            //  Check  for  init.  button
167            if  (!(PORTC  &  0x01))  {
168                //  Active
169                vscp_initbtncnt++;
170            }
171            else  {
172                vscp_initbtncnt  =  0;
173            }
174  
175            //  Status  LED
176            vscp_statuscnt++;
177            if  (  (  VSCP_LED_BLINK1  ==
178                     vscp_initledfunc)  &&  (vscp_statuscnt &rsaquo   100))  {
179                if  (  PORTC  &  0x02  )  {
180                    PORTC  &=  ~0x02;
181                }
182                else  {
183                    PORTC  |=  0x02;
184                }
185                vscp_statuscnt  =  0;
186            }
187            else  if  (VSCP_LED_ON  ==  vscp_initledfunc)  {
188                PORTC  |=  0x02;
189                vscp_statuscnt  =  0;
190            }
191            else  if  (VSCP_LED_OFF  ==  vscp_initledfunc)  {
192                PORTC  &=  ~0x02;
193                vscp_statuscnt  =  0;
194            }
195  
196            INTCONbits.TMR0IF  =  0;  //  Clear  Timer0  Interrupt  Flag
197  
198        }  //  Timer  &  button
199  
200        //  Check  ADC
201        if  (  PIR1bits.ADIF  )  {
202  
203            switch  (0x3C  &  ADCON0)  {
204  
205                case  SELECT_ADC_TEMP0:
206                    //  Read  conversion
207                    adc_low  =  ADRESH;
208                    adc_high  =  ADRESL;
209                    //  Start  new  conversion
210                    ADCON0  =  SELECT_ADC_TEMP0  +  1;
211                    break;
212  
213            }
214  
215            //  Start  conversion
216            ConvertADC();
217  
218            PIR1bits.ADIF  =  0;  //  Reset  interrupt  flag
219  
220        }  //  ADC
221  
222        return;
223   }

360  
361   void  doWork(void)
362   {
363        uint8_t  i,  j;
364        uint16_t  B;
365        double  resistance;
366        double  Rinf;
367        double  temp;
368        double  v;
369        double  calVoltage;
370  
371        calVoltage  =  ((uint16_t)  eeprom_read(EEPROM_CALIBRATED_VOLTAGE_MSB)*256  +
372                        eeprom_read(EEPROM_CALIBRATED_VOLTAGE_LSB));
373  
374        //  Check  if  there  are  new  adc  values  to
375        //  convert  to  temperatures
376  
377        //  Use  B-constant
378        //  ==============
379        //  http://en.wikipedia.org/wiki/Thermistor
380        //  R1  =  (R2V  -  R2V2)  /  V2   R2=  10K,  V  =  5V,   V2  =  adc  *  voltage/1024
381        //  T  =  B  /  ln(r/Rinf)
382        //  Rinf  =  R0  e  (-B/T0),  R0=10K,  T0  =  273.15  +  25  =  298.15
383        B  =  (uint16_t)eeprom_read(2  *  i  +  EEPROM_B_CONSTANT0_MSB)*256  +
384                            eeprom_read(2  *  i  +  EEPROM_B_CONSTANT0_LSB);
385               
386        Rinf  =  10000.0  *  exp(B  /  -298.15);
387        //itemp  =  Rinf  *  10000;
388        v  =  5.0  *  (double)  (  (  adc_high &lsaquo &lsaquo   8  )  +  adc_low)  /  1025;  //  TODO  12-bit  resolution
389        //itemp  =  v  *  100;
390        resistance  =  (calVoltage  -  10000.0  *  v)  /  v;
391        //itemp  =  r;
392        temp  =  ((double)  B)  /  log(resistance  /  Rinf);
393        //itemp  =  log(r/Rinf);
394        temp  -=  273.15;  //  Convert  Kelvin  to  Celsius            
395  
396        current_temp  =  (current_temp  + 
397                                ((long)(temp  *  100)  + 
398                                getCalibrationValue(i)  )  )  /  2;
399  
400   }
