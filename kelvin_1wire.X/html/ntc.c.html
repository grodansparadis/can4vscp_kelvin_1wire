41   double  Celsius2Fahrenheit(double  tc)
42   {
43        return  (  (9  *  tc  +  16000)  /  5);
44   }

47   //  Fahrenheit2Celsius
48   //
49  
50   double  Fahrenheit2Celsius(double  tf)
51   {
52        return  (  (5  *  tf  -  16000)  /  9);
53   }
